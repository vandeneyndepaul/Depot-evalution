
DROP DATABASE IF EXISTS APPLICATION_SAV_DIRUY;

CREATE DATABASE APPLICATION_SAV_DIRUY;

USE APPLICATION_SAV_DIRUY;

CREATE TABLE Client (
	Client_Id		INt,
	client_Nom		VARCHAR(40) NOT NULL,
	client_Prenom	VARCHAR(40) NULL,
	Client_Adress1	VARCHAR(40) NOT NULL,
	Client_Adress2	VARCHAR(40) NULL,
	Client_Code_Post	VARCHAR(5) NOT NULL,
	Client_ville	VARCHAR(40) NOT NULL
		
);


CREATE TABLE SAV(
      Client_Id		INt,
      SAV_Nom		VARCHAR(40) NOT NULL,
		SAV_Prenom	VARCHAR(40) NULL,
      SAV_Adress1	VARCHAR(40) NOT NULL,
	  	SAV_Adress2	VARCHAR(40) NULL,
	   SAV_Code_Post	VARCHAR(5) NOT NULL,
	   SAV_ville	VARCHAR(40) NOT NULL,
	   SAV_Probleme TEXT NOT NULL,
	   Dt_Crea DATETIME NOT NULL
);
